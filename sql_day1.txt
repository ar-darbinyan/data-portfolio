-- SQL Day 1 (Basics)

-- Example 1:Find the movies released in the years between 2000 and 2010 
SELECT title, year FROM movies
WHERE year BETWEEN 2000 AND 2010;

-- Example 2:Find all the Toy Story movies 
SELECT title, director FROM movies 
WHERE title LIKE "Toy Story%";

-- Example 3:Find all the movies (and director) not directed by John Lasseter
SELECT title, director FROM movies 
WHERE director != "John Lasseter";

-- Example 4:Find all the WALL-* movies
SELECT * FROM movies 
WHERE title LIKE "WALL-_";
